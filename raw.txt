    // writes current numbers to a txt file. just a failsafe so we don't start at 0 if closed
    private static void saveCounterState() {
        try {
            PrintWriter writer = new PrintWriter(BACKUP_FILE, "UTF-8");
            writer.println(counters.get("Payments"));
            writer.println(counters.get("Registration"));
            writer.println(counters.get("Consultation"));
            writer.close();
        } catch (Exception e) { System.out.println("Failed to save state."); }
    }

    // reads the txt file on startup to restore ticket numbers
    private static void loadCounterState() {
        try {
            File backup = new File(BACKUP_FILE);
            if (backup.exists()) {
                Scanner scanner = new Scanner(backup);
                if(scanner.hasNextLine()) counters.put("Payments", Integer.parseInt(scanner.nextLine().trim()));
                if(scanner.hasNextLine()) counters.put("Registration", Integer.parseInt(scanner.nextLine().trim()));
                if(scanner.hasNextLine()) counters.put("Consultation", Integer.parseInt(scanner.nextLine().trim()));
                scanner.close();
            }
        } catch (Exception e) { System.out.println("Failed to load state. Starting at 001."); }
    }

    // core logic: makes a new ticket, increments counter, adds to line, and logs the time
    public static synchronized String generateNextTicket(String category) {
        int num = counters.get(category);
        queues.get(category).add(num);
        counters.put(category, num + 1);

        saveCounterState();

        String prefix = category.substring(0, 1).toUpperCase();
        String ticket = prefix + "-" + String.format("%03d", num);
        ticketTimestamps.put(ticket, System.currentTimeMillis());
        return ticket;
    }

    // helper to make the 3 rows in the dashboard for each department
    private JPanel createControlRow(String category, String prefix) {
        JPanel row = new JPanel(new BorderLayout());
        row.setOpaque(false);
        row.setBorder(BorderFactory.createMatteBorder(0, 0, 1, 0, new Color(255, 255, 255, 15)));

        JLabel catLabel = new JLabel(category);
        catLabel.setFont(new Font("Segoe UI", Font.BOLD, 22));
        catLabel.setForeground(Color.WHITE);

        JPanel btnPanel = new JPanel(new FlowLayout(FlowLayout.RIGHT, 15, 10));
        btnPanel.setOpaque(false);

        StyledButton btnGen = new StyledButton("Generate", ACCENT_GREEN);
        StyledButton btnNext = new StyledButton("Call Next", ACCENT_ORANGE);

        // local admin gen button action
        btnGen.addActionListener(e -> {
            String tkt = generateNextTicket(category);
            showToast("Generated " + tkt, ACCENT_GREEN);
        });

        // call next button action: pops queue, updates analytics, plays sound
        btnNext.addActionListener(e -> {
            Queue<Integer> q = queues.get(category);
            if (q.isEmpty()) {
                showToast("Queue is empty!", ADMIN_RED);
            } else {
                btnNext.setEnabled(false);
                Timer cooldown = new Timer(1500, evt -> btnNext.setEnabled(true));
                cooldown.setRepeats(false); cooldown.start(); // 1.5s delay to stop spam clicking

                int nextNum = q.poll();
                String tkt = prefix + "-" + String.format("%03d", nextNum);
                currentServing.put(category, tkt);

                if (ticketTimestamps.containsKey(tkt)) {
                    long waitTime = System.currentTimeMillis() - ticketTimestamps.get(tkt);
                    totalWaitTimeMillis += waitTime;
                    totalServedToday++;

                    categoryWaitTimeMillis.put(category, categoryWaitTimeMillis.get(category) + waitTime);
                    categoryServedToday.put(category, categoryServedToday.get(category) + 1);
                    updateAnalyticsUI();
                }

                String time = timeFormat.format(new Date());
                String log = "[" + time + "] " + category + " called " + tkt;
                historyLog.add(log);
                historyModel.add(0, log);
                playAnnouncement(category, tkt);
            }
        });

        btnPanel.add(btnGen); btnPanel.add(btnNext);
        row.add(catLabel, BorderLayout.WEST); row.add(btnPanel, BorderLayout.EAST);
        return row;
    }

    // refreshes the labels in the analytics tab when a new person is served
    private void updateAnalyticsUI() {
        lblTotalServed.setText(String.valueOf(totalServedToday));
        if (totalServedToday > 0) {
            long avgMillis = totalWaitTimeMillis / totalServedToday;
            long mins = (avgMillis / 1000) / 60;
            long secs = (avgMillis / 1000) % 60;
            lblAvgWaitTime.setText(mins + "m " + secs + "s");
        }
    }

    // dumps the history log into a nice excel csv file
    private void exportToCSV() {
        if (historyLog.isEmpty()) {
            showToast("No data to export yet.", ADMIN_RED);
            return;
        }
        try {
            String date = dateFormat.format(new Date());
            File file = new File("QueueCare_Report_" + date + ".csv");
            PrintWriter pw = new PrintWriter(file);
            pw.println("Timestamp,Category,Action");
            for (String log : historyLog) {
                String cleanLog = log.replace("[", "").replace("]", ",");
                pw.println(cleanLog.replace(" called ", ",Called "));
            }
            pw.close();
            showToast("Saved to: " + file.getAbsolutePath(), ACCENT_GREEN);
        } catch (Exception ex) {
            showToast("Export failed.", ADMIN_RED);
        }
    }

    // helper to build the red danger buttons
    private StyledButton createResetButton(String text, String category) {

        btn.addActionListener(e -> {
            if (JOptionPane.showConfirmDialog(this, "Reset " + category + " to 001?", "Warning", JOptionPane.YES_NO_OPTION) == JOptionPane.YES_OPTION) {
                queues.get(category).clear();
                counters.put(category, 1);
                currentServing.put(category, "--");
                categoryServedToday.put(category, 0);
                categoryWaitTimeMillis.put(category, 0L);
                saveCounterState(); // overwrite the txt file so it starts fresh
                showToast(category + " Reset to 001", ADMIN_RED);
            }
        }); return btn;
    }

    // struct to hold audio output choices
    class MixerItem {
        Mixer.Info info; String name;
        public MixerItem(Mixer.Info info, String name) { this.info = info; this.name = name; }
        @Override public String toString() { return name; }
    }

    // scans the PC for speakers and headphones
    private Vector<MixerItem> getAudioDevices() {
        Vector<MixerItem> items = new Vector<>(); items.add(new MixerItem(null, "System Default Output"));
        for (Mixer.Info m : AudioSystem.getMixerInfo()) {
            if (AudioSystem.getMixer(m).getTargetLineInfo().length > 0) items.add(new MixerItem(m, m.getName()));
        } return items;
    }

    // the logic that actually plays the "Ding!" or the custom wav file
    private void playAnnouncement(String category, String ticket) {
        MixerItem selectedDevice = (MixerItem) audioDeviceDropdown.getSelectedItem();
        String customSoundPath = customSounds.get(category);
        new Thread(() -> { // put in a thread so the UI doesn't freeze while audio plays
            try {
                if (customSoundPath != null && !customSoundPath.isEmpty() && new File(customSoundPath).exists()) {
                    try {
                        AudioInputStream ais = AudioSystem.getAudioInputStream(new File(customSoundPath));
                        Clip clip = (selectedDevice.info != null) ? AudioSystem.getClip(selectedDevice.info) : AudioSystem.getClip();
                        clip.open(ais); clip.start(); return;
                    } catch (UnsupportedAudioFileException e) { SwingUtilities.invokeLater(() -> showToast("Java requires .wav format.", ADMIN_RED)); }
                }
                // fallback chime sound using raw sine wave
                byte[] buf = new byte[8000]; for (int i = 0; i < buf.length; i++) buf[i] = (byte) (Math.sin(i / (8000.0 / 440) * 2.0 * Math.PI) * 127.0);
                AudioFormat af = new AudioFormat(8000f, 8, 1, true, false);
                SourceDataLine sdl = (selectedDevice.info != null) ? AudioSystem.getSourceDataLine(af, selectedDevice.info) : AudioSystem.getSourceDataLine(af);
                sdl.open(af); sdl.start(); sdl.write(buf, 0, buf.length); sdl.drain(); sdl.close();
                // trigger text to speech depending if on windows or mac
                String os = System.getProperty("os.name").toLowerCase();
                if (os.contains("win")) Runtime.getRuntime().exec(new String[]{"PowerShell", "-Command", "Add-Type -AssemblyName System.Speech; (New-Object System.Speech.Synthesis.SpeechSynthesizer).Speak('Ticket " + ticket + ", please proceed to " + category + "');"});
                else if (os.contains("mac")) Runtime.getRuntime().exec(new String[]{"say", "Ticket " + ticket + ", please proceed to " + category});
            } catch (Exception ex) { ex.printStackTrace(); }
        }).start();
    }

    // NEW FIX: This makes sure the app ALWAYS finds the 'web' folder,
    // whether running from IntelliJ or from a standalone .jar file
    private static File getWebFolder() {
        try {
            // Figure out exactly where the code/jar is physically saved on the computer
            File location = new File(QueueCare.class.getProtectionDomain().getCodeSource().getLocation().toURI());
            if (location.isFile()) {
                // If it's a file (a .jar), look in the folder holding the .jar
                return new File(location.getParentFile(), "web");
            } else {
                // If we are in IntelliJ, just use the standard relative path
                return new File("web");
            }
        } catch (Exception e) {
            return new File("web"); // Emergency fallback
        }
    }

    // grabs html files from our external "web" folder so we don't have messy strings in java
    private static String readHtmlFile(String fileName) {
        try {
            File htmlFile = new File(getWebFolder(), fileName);
            return Files.readString(htmlFile.toPath(), StandardCharsets.UTF_8);
        } catch (IOException e) {
            System.err.println("CRITICAL ERROR: Could not read file: " + fileName);
            return "<!DOCTYPE html><html><body style='background:#0f172a; color:white; font-family:sans-serif; text-align:center; padding-top:50px;'>" +
                    "<h2>Error 404: File Not Found</h2>" +
                    "<p>The server could not locate <b>web/" + fileName + "</b>.</p>" +
                    "<p>Please ensure the 'web' folder is located next to your .jar file!</p></body></html>";
        }
    }

    // --- SECURE TICKET VERIFICATION API ---
    // handles when someone tries to recover a ticket. makes sure it actually exists and wasn't skipped
    static class VerifyApiHandler implements HttpHandler {
        @Override public void handle(HttpExchange t) throws IOException {
            String query = t.getRequestURI().getQuery();
            String res = "{\"valid\": false, \"reason\": \"Invalid request format.\"}";

            try {
                if (query != null && query.startsWith("tkt=")) {
                    String tkt = query.substring(4).split("&")[0].toUpperCase();
                    if (tkt.contains("-")) {
                        String[] parts = tkt.split("-");
                        String prefix = parts[0];
                        int num = Integer.parseInt(parts[1]);
                        String cat = prefix.equals("P") ? "Payments" : prefix.equals("R") ? "Registration" : prefix.equals("C") ? "Consultation" : "";

                        if (!cat.isEmpty()) {
                            boolean isServing = tkt.equals(currentServing.get(cat));
                            boolean isWaiting = queues.get(cat).contains(num);

                            if (isServing || isWaiting) {
                                res = "{\"valid\": true, \"cat\": \"" + cat + "\"}";
                            } else {
                                if (num >= counters.get(cat)) {
                                    res = "{\"valid\": false, \"reason\": \"This ticket has not been generated yet.\"}";
                                } else {
                                    res = "{\"valid\": false, \"reason\": \"This ticket has already been served and is no longer valid.\"}";
                                }
                            }
                        } else {
                            res = "{\"valid\": false, \"reason\": \"Invalid department prefix.\"}";
                        }
                    }
                }
            } catch (Exception e) {}

            byte[] bytes = res.getBytes(StandardCharsets.UTF_8);
            t.getResponseHeaders().set("Content-Type", "application/json");
            t.sendResponseHeaders(200, bytes.length);
            OutputStream os = t.getResponseBody(); os.write(bytes); os.close();
        }
    }

    // --- TICKET CANCELLATION API ---
    // lets client drop out of line, removes their number from the background array
    static class CancelApiHandler implements HttpHandler {
        @Override public void handle(HttpExchange t) throws IOException {
            String query = t.getRequestURI().getQuery();
            String res = "{\"status\": \"success\"}";
            try {
                if (query != null) {
                    String[] params = query.split("&");
                    String cat = "";
                    int num = -1;
                    for (String p : params) {
                        if (p.startsWith("cat=")) cat = p.substring(4);
                        if (p.startsWith("num=")) num = Integer.parseInt(p.substring(4));
                    }
                    if (!cat.isEmpty() && num != -1 && queues.containsKey(cat)) {
                        queues.get(cat).remove(Integer.valueOf(num)); // drop from memory
                    }
                }
            } catch (Exception e) {
                res = "{\"error\": \"failed\"}";
            }
            byte[] bytes = res.getBytes(StandardCharsets.UTF_8);
            t.getResponseHeaders().set("Content-Type", "application/json");
            t.sendResponseHeaders(200, bytes.length);
            OutputStream os = t.getResponseBody(); os.write(bytes); os.close();
        }
    }

    // --- TICKET GENERATOR API ---
    // when a browser clicks request ticket, this handles generating it while tracking their IP so they cant spam it
    static class GenerateApiHandler implements HttpHandler {
        @Override public void handle(HttpExchange t) throws IOException {
            String query = t.getRequestURI().getQuery();
            String res = "{\"error\": \"Invalid\"}";
            int code = 200;

            String userIP = t.getRemoteAddress().getAddress().getHostAddress();
            long currentTime = System.currentTimeMillis();

            try {
                if (ipCooldowns.containsKey(userIP) && (currentTime - ipCooldowns.get(userIP)) < 10000) {
                    res = "{\"error\": \"Cooldown active. Please wait 10 seconds.\"}";
                    code = 429;
                } else if (query != null) {
                    String[] params = query.split("&");
                    String cat = "";
                    for (String p : params) {
                        if (p.startsWith("cat=")) cat = p.substring(4);
                    }
                    if (!cat.isEmpty() && queues.containsKey(cat)) {
                        ipCooldowns.put(userIP, currentTime); // log ip so they cant request again immediately
                        res = "{\"ticket\": \"" + generateNextTicket(cat) + "\"}";
                    }
                }
            } catch (Exception e) {}

            byte[] bytes = res.getBytes(StandardCharsets.UTF_8);
            t.getResponseHeaders().set("Content-Type", "application/json");
            t.sendResponseHeaders(code, bytes.length);
            OutputStream os = t.getResponseBody(); os.write(bytes); os.close();
        }
    }

    // --- LIVE FEED API ---
    // broadcasts the active serving numbers to all the connected web clients as JSON
    static class ApiHandler implements HttpHandler {
        @Override public void handle(HttpExchange t) throws IOException {
            StringBuilder json = new StringBuilder("{"); String[] cats = {"Payments", "Registration", "Consultation"};
            for (int i = 0; i < cats.length; i++) {

                long served = categoryServedToday.getOrDefault(c, 0);
                long waitMillis = categoryWaitTimeMillis.getOrDefault(c, 0L);
                String avgStr = "0m 0s";
                if (served > 0) {
                    long avg = waitMillis / served;
                    avgStr = (avg/1000/60) + "m " + (avg/1000%60) + "s";

                        .append("\"waiting\":").append(queues.get(c).size()).append(",")
                        .append("\"avgWait\":\"").append(avgStr).append("\"")
                        .append("}");
                if (i < cats.length - 1) json.append(",");
            }
            json.append("}");

            byte[] bytes = json.toString().getBytes(StandardCharsets.UTF_8);
            t.getResponseHeaders().set("Content-Type", "application/json");

            try (OutputStream os = t.getResponseBody(); FileInputStream fis = new FileInputStream(file)) {
                // read the file in chunks so it doesn't crash RAM on big videos
                byte[] buffer = new byte[64 * 1024]; int count; while ((count = fis.read(buffer)) != -1) os.write(buffer, 0, count);
            }
        }
    }

    // main entrypoint. kicks off the local server thread and boots the UI
    public static void main(String[] args) {
        startWebServer(); SwingUtilities.invokeLater(() -> new QueueCare().setVisible(true));
    }
